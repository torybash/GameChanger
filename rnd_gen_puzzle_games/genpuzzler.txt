BasicGame
	SpriteSet
		avatar > MovingAvatar img=avatar
		gen1 > Passive img=fire
		gen2 > Passive img=shovel
	InteractionSet
		gen1 wall > stepBack
		gen1 gen2 > bounceForward scoreChange=-3
		gen1 avatar > transformTo stype=gen2
		avatar gen1 > undoAll scoreChange=6
		avatar gen1 > killSprite
		gen1 gen2 > stepBack
		gen2 wall > bounceForward
		gen2 avatar > bounceForward scoreChange=-2
		avatar gen2 > undoAll
		gen1 gen1 > undoAll
		gen1 gen1 > killSprite
		gen1 avatar > transformTo stype=gen2
		gen2 gen2 > undoAll scoreChange=-4
		avatar wall > transformTo stype=avatar
		avatar wall > stepBack
		gen1 wall > stepBack
		gen2 wall > stepBack
	LevelMapping
		$ > gen1 
		% > gen2 
	TerminationSet
		SpriteCounter limit=0 stype=gen2 win=TRUE
