BasicGame
	SpriteSet
		avatar > MovingAvatar img=avatar
		gen1 > Immovable img=bullet
		gen2 > Immovable img=sword
	InteractionSet
		avatar gen2 > bounceForward scoreChange=3
		avatar gen2 > undoAll
		avatar EOS > killSprite
		gen1 avatar > transformTo stype=gen1
		avatar wall > stepBack
		gen1 wall > stepBack
		gen2 wall > stepBack
	LevelMapping
		$ > gen1 
		% > gen2 
	TerminationSet
		MultiSpriteCounter limit=0 stype1=gen1 stype2=gen2 win=TRUE
