BasicGame
	SpriteSet
		avatar > MovingAvatar img=avatar
		gen1 > Passive img=hole
		gen2 > Portal stype=gen3 img=camel
		gen3 > Passive img=fire
	InteractionSet
		avatar gen3 > transformTo stype=avatar
		gen1 avatar > bounceForward
		avatar gen1 > transformTo stype=avatar scoreChange=-1
		gen2 avatar > bounceForward
		avatar gen3 > stepBack
		gen2 avatar > bounceForward
		gen1 gen3 > killSprite scoreChange=5
		gen1 gen3 > undoAll
		gen1 wall > stepBack
		gen1 gen1 > undoAll scoreChange=5
		gen2 gen2 > undoAll
		gen1 avatar > killSprite
		avatar gen2 > bounceForward scoreChange=10
		avatar wall > stepBack
		gen1 wall > stepBack
		gen2 wall > stepBack
		gen3 wall > stepBack
	LevelMapping
		$ > gen1 
		% > gen2 
		& > gen3 
	TerminationSet
		SpriteCounter limit=0 stype=gen1 win=TRUE
